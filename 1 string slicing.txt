 slicing = create a substring by extracting elements from another string 
           --> indexing[]    start:stop:step]
           --> slice()       start:stop:step]
           
           
           

name = "John Kowalski"

first_name = name[:3]       #output: [0:4]       --> "John"      --> first char is always starting from 0 not from 1, -> :4 means from first char to fifth char
last_name = name[4:]        #output: [5:end]     --> "Kowalski"  --> 5:  it means from sixth char till the last char
funky_name = name[::2]      #output: [0:end:2]   --> "Jh oasi"      --> from first to the last char, every other char
reversed_name = name[::-1]  #output: [0:end:-1]  --> "edoC orB"  --> from first to the last char, reversed order
